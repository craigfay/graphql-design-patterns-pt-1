```graphql title="Filter Types"
input StringFilter {
  eq: String
  startsWith: String
  endsWith: String
  contains: String
  istartsWith: String # case insensitive
  iendsWith: String
  icontains: String
  in: [String]
}

input IntFilter {
  eq: Int
  gt: Int
  lt: Int
  gte: Int
  lte: Int
  in: [Int]
}

input DateTimeFilter {
  eq: DateTime
  gt: DateTime
  lt: DateTime
  gte: DateTime
  lte: DateTime
  in: [DateTime]
}

input ArticleSearchFilters {
  id: IntFilter
  createdAt: DateTimeFilter
  updatedAt: DateTimeFilter
  title: StringFilter
  isPublished: BooleanFilter
}

input ArticleSearchCriteria {
  filters: ArticleSearchFilters
  first: Int,
  last: Int,
  after: String,
  before: String,
}

query {
  articleList(criteria: {
    filters: {
      createdAt: { gt: "2022-10-11T20:09:40.326Z" },
      teaserText: { icontains: "ChatGPT" },
    }
  }) {
    payload {
      nodes {
        id
        title
        teaserText
        createdAt
      }
    }
  }
}
```
